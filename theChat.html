  form.parentNode.style.display = "none";

      // Check if the chatbot has finished responding

        mutation.target.classList.contains("outgoing") &&
        !mutation.target.classList.contains("sending")
      ) {
        // Scroll to the bottom of the conversation
        conversation.scrollTop = conversation.scrollHeight;
      }
    });
  });
